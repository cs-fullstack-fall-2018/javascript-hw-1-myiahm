How do you evaluate the expressions below? Next to each console log type out the resulting string.

var a = 5, b = 9; 
console.log("a = %d, b = %d\n", a, b); 
5
9
console.log("a&b = %d\n", a&b);
1
console.log("a|b = %d\n", a|b);
13
console.log("b<<1 = %d\n", b<<1);
18
console.log("b>>1 = %d\n", b>>1);
4
